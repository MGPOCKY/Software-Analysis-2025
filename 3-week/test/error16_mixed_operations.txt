// 타입 오류: 복합적인 타입 불일치
process(ptr, size) {
  var i, sum;
  i = 0;
  sum = ptr;  // 오류: sum에 pointer 할당
  while (i < size) {
    sum = sum + *(ptr + i);  // 오류: pointer + int
    i = i + 1;
  }
  return sum;
}

main() {
  var data, result;
  data = alloc 5;
  result = process(data, 5);
  return result;
}
