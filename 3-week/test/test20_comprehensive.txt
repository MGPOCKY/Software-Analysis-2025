add_two_numbers(x, y) {
  var result;
  result = x + y;
  return result;
}

process_data(ptr_a, ptr_b) {
  var val_a, val_b, sum;
  val_a = *ptr_a;
  val_b = *ptr_b;
  sum = add_two_numbers(val_a, val_b);
  return sum;
}

main() {
  var x, y, ptr_x, ptr_y, result;
  x = 10;
  y = 20;
  ptr_x = &x;
  ptr_y = &y;
  result = process_data(ptr_x, ptr_y);
  output result;
  return result;
}
