main() {
  var counter, target, ptr, current_val, sum;

  counter = 0;
  target = 3;
  sum = 0;

  ptr = &counter;

  while (*ptr == counter) {
    current_val = counter * 2;
    sum = sum + current_val;
    counter = counter + 1;

    if (counter == target) {
      *ptr = 999; // 루프 종료 조건
    }
  }

  return sum;
}
