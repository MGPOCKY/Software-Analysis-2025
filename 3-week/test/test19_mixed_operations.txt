main() {
  var num1, num2, ptr, temp, sum, product, final_result;

  num1 = 5;
  num2 = 3;
  ptr = &num1;

  // 포인터를 통한 값 변경
  *ptr = 7;

  // 혼합 연산: 포인터 역참조, 산술 연산, 비교 연산
  temp = *ptr + num2;
  sum = temp * 2;

  // 조건문과 할당
  if (sum > 15) {
    product = sum * 2;
  } else {
    product = sum;
  }

  final_result = product;
  output final_result;
  return final_result;
}
